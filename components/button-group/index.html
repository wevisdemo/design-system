Button Group

Button container with spacing. Align hotizontally on desktop and vertically on mobile.

import '@wevisdemo/ui/styles/button.css';
import '@wevisdemo/ui/styles/button-group.css';
import WvButton from '@wevisdemo/ui/{react,vue2,svelte}/button';
import WvButtonGroup from '@wevisdemo/ui/{react,vue2,svelte}/button-group';
<WvButtonGroup>
  <WvButton>A</WvButton>
  <WvButton>B</WvButton>
  <WvButton>C</WvButton>
</WvButtonGroup>
